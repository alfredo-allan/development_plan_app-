// utils/PermissionsManager.kt
@Composable
fun RequestStoragePermission() {
    val permissionState = rememberPermissionState(
        android.Manifest.permission.READ_EXTERNAL_STORAGE
    )
    
    if (!permissionState.hasPermission) {
        LaunchedEffect(Unit) {
            permissionState.launchPermissionRequest()
        }
    }
}
3.2 Navegador de Arquivos

// ui/screens/FileBrowserScreen.kt
@Composable
fun FileBrowserScreen(
    onFileSelected: (File) -> Unit,
    onFolderSelected: (File) -> Unit
) {
    var currentPath by remember { mutableStateOf(Environment.getExternalStorageDirectory()) }
    
    LazyColumn {
        items(fileList) { file ->
            FileItem(
                file = file,
                onClick = {
                    if (file.isDirectory) {
                        currentPath = file
                        onFolderSelected(file)
                    } else {
                        onFileSelected(file)
                    }
                }
            )
        }
    }
}

@Composable
fun ShortcutItem(

    onClick: () -> Unit
) {
    Column(
        horizontalAlignment = Alignment.CenterHorizontally,
        modifier = Modifier
            .clickable { onClick() }
            .padding(8.dp)
    ) {
        Icon(
            imageVector = getIconForType(shortcut.type),
            contentDescription = shortcut.title
        )
        Text(
            text = shortcut.title,
            fontSize = 12.sp,
            textAlign = TextAlign.Center
        )
    }
}

@Composable
fun PdfViewerScreen(
    pdfPath: String,
    onPageChanged: (Int) -> Unit,
    initialPage: Int = 0
) {
    val pdfView = remember { PDFView(context, null) }
    
    AndroidView(
        factory = { context ->
            pdfView.fromFile(File(pdfPath))
                .defaultPage(initialPage)
                .onPageChange { page, _ ->
                    onPageChanged(page)
                }
                .load()
            pdfView
        }
    )
}
